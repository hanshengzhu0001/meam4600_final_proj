\begin{frame}{Scope}
\small
\begin{itemize}
  \item Benchmark: nonlinear elliptic inverse problem, \(N_x=63\), \(\kappa=50\), periodic BC.
  \item Task: recover full forcing \(u\) from partial observations of \(v\).
  \item Known vs unknown: only sparse entries of \(v\) are observed (10\% mask); full \(u\) and full \(v\) are both unknown and must be inferred.
  \item Evaluation: full table over schedules/orders with \(128\) samples and \(3\) observation seeds.
  \item Metrics (all lower is better): \texttt{u\_error}, \texttt{v\_error}, \texttt{obs\_error}, \texttt{posterior\_quality}, \texttt{physical\_consistency}, \texttt{runtime}, \texttt{trajectory\_stability}, \texttt{mmse}, \texttt{smse}, \texttt{fpd}.
\end{itemize}
\end{frame}

\begin{frame}{Metric Definitions Used in This Study}
\scriptsize
\begin{tabularx}{\linewidth}{@{}p{0.18\linewidth}p{0.77\linewidth}@{}}
\toprule
\textbf{Metric} & \textbf{Definition / Interpretation} \\
\midrule
\texttt{u\_error} & Relative \(L^2\) error of inferred full forcing \(u\) (primary inverse target). \\
\texttt{v\_error} & Relative \(L^2\) error of inferred full state \(v\); \(v\) is only partially observed, so this captures unobserved-region reconstruction. \\
\texttt{obs\_error} & RMS error on observed entries of \(v\) only (masked observation fit). \\
\texttt{mmse} & Mean-squared difference between generated and reference sample means in joint state space \([u;v]\). \\
\texttt{smse} & Mean-squared difference between generated and reference sample standard deviations in joint state space \([u;v]\). \\
\texttt{fpd} & Fr\'echet-style distribution distance using empirical Gaussian moments of generated vs reference samples. \\
\texttt{ce\_ic}/\texttt{ce\_bc}/\texttt{ce\_cl} & Constraint errors for initial, boundary, and conservation-law constraints. Non-applicable constraints are recorded as NaN and displayed as N/A in tables. \\
\bottomrule
\end{tabularx}
\end{frame}
